## 第 06 章：多表關聯 JOIN {#第06章}

**6-1**
```sql
SELECT
    p.title         AS 書名,
    c.category_name AS 分類名稱
FROM products p
JOIN categories c ON p.category_id = c.category_id
ORDER BY c.category_name, p.title;
```

**6-2**
```sql
SELECT
    o.order_id    AS 訂單編號,
    c.name        AS 顧客姓名,
    o.order_date  AS 訂購日期
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id
ORDER BY o.order_date;
```

**6-3**
```sql
SELECT
    p.title      AS 書名,
    oi.quantity  AS 數量
FROM customers c
JOIN orders o      ON c.customer_id = o.customer_id
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p    ON oi.product_id = p.product_id
WHERE c.name = '陳小明';
```

**6-4**
```sql
SELECT c.name AS 從未下單的顧客
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;
```

**6-5**
```sql
SELECT
    c.name              AS 顧客姓名,
    COUNT(o.order_id)   AS 訂單數量
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.name
ORDER BY 訂單數量 DESC;
```

**6-6**（挑戰）
```sql
SELECT
    p.title         AS 書名,
    a.name          AS 作者,
    cat.category_name AS 分類,
    COUNT(oi.item_id) AS 購買次數
FROM products p
JOIN authors a       ON p.author_id   = a.author_id
JOIN categories cat  ON p.category_id = cat.category_id
LEFT JOIN order_items oi ON p.product_id = oi.product_id
GROUP BY p.product_id, p.title, a.name, cat.category_name
ORDER BY 購買次數 DESC;
```

## 第 07 章：子查詢 {#第07章}

**7-1**
```sql
SELECT title, price
FROM products
WHERE price > (SELECT AVG(price) FROM products)
ORDER BY price DESC;
```

**7-2**
```sql
SELECT title
FROM products
WHERE category_id = (
    SELECT category_id
    FROM categories
    WHERE category_name = '科幻'
);
```

**7-3**
```sql
SELECT title
FROM products
WHERE product_id NOT IN (
    SELECT DISTINCT product_id
    FROM order_items
);
```

**7-4**
```sql
SELECT name
FROM customers c
WHERE EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id
      AND o.status = 'completed'
);
```

**7-5**
```sql
SELECT
    title,
    price,
    ROUND(price - (SELECT AVG(price) FROM products), 0) AS 與均價差距
FROM products
ORDER BY price DESC;
```

**7-6**（挑戰）
```sql
SELECT p.title, p.category_id, p.price
FROM products p
JOIN (
    SELECT category_id, MAX(price) AS max_price
    FROM products
    GROUP BY category_id
) AS cat_max
  ON p.category_id = cat_max.category_id
 AND p.price = cat_max.max_price
ORDER BY p.category_id;
```

## 第 08 章：新增、修改、刪除 {#第08章}

INSERT INTO products (title, author_id, category_id, price, stock, published_year)

SELECT * FROM products WHERE title = '設計的法則';

**8-2**
```sql
-- 先確認
SELECT title, stock FROM products WHERE title = '挪威的森林';

UPDATE products
SET stock = stock + 5
WHERE title = '挪威的森林';

-- 驗證
SELECT title, stock FROM products WHERE title = '挪威的森林';
```

**8-3**
```sql
-- 先確認
SELECT title, price FROM products WHERE stock = 0;

-- 修改
UPDATE products
SET price = ROUND(price * 0.9, 0)
WHERE stock = 0;
```

**8-4**
```sql
-- ① 先確認
SELECT * FROM orders WHERE status = 'cancelled';

-- ② 刪除
DELETE FROM orders WHERE status = 'cancelled';

-- ③ 驗證
SELECT COUNT(*) FROM orders WHERE status = 'cancelled';
-- 應回傳 0
```

**8-5**（挑戰）
```sql
BEGIN;

-- 新增訂單
INSERT INTO orders (customer_id, order_date, status)
VALUES (3, DATE('now'), 'pending');

-- 取得剛插入的訂單 ID
-- 新增訂單明細
INSERT INTO order_items (order_id, product_id, quantity, unit_price)
VALUES (last_insert_rowid(), 2, 1, 380);

COMMIT;

-- 驗證
SELECT o.order_id, o.order_date, oi.product_id, oi.quantity
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
WHERE o.customer_id = 3
ORDER BY o.order_id DESC
LIMIT 1;
```

## 第 09 章：建立與管理表格 {#第09章}

**9-1**
```sql
CREATE TABLE IF NOT EXISTS reviews (
    review_id   INTEGER PRIMARY KEY AUTOINCREMENT,
    product_id  INTEGER NOT NULL REFERENCES products(product_id),
    customer_id INTEGER NOT NULL REFERENCES customers(customer_id),
    rating      INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment     TEXT,
    created_at  TEXT    NOT NULL DEFAULT (DATETIME('now'))
);
```

## 綜合挑戰 {#綜合}

**A-1**
```sql
SELECT
    c.name                               AS 顧客姓名,
    SUM(oi.quantity * oi.unit_price)     AS 消費總額
FROM customers c
JOIN orders o       ON c.customer_id = o.customer_id
JOIN order_items oi ON o.order_id    = oi.order_id
WHERE o.status = 'completed'
GROUP BY c.customer_id, c.name
ORDER BY 消費總額 DESC;
```

**A-2**
```sql
SELECT
    p.title      AS 書名,
    COUNT(*)     AS 購買次數
FROM customers c
JOIN orders o       ON c.customer_id = o.customer_id
JOIN order_items oi ON o.order_id    = oi.order_id
JOIN products p     ON oi.product_id = p.product_id
WHERE c.city = '台北'
GROUP BY p.product_id, p.title
ORDER BY 購買次數 DESC
LIMIT 1;
```

**A-3**（假設已建立 `reviews` 表格並有資料）
```sql
SELECT
    p.title          AS 書名,
    ROUND(AVG(r.rating), 2) AS 平均評分,
    COUNT(*)         AS 評論數
FROM products p
JOIN reviews r ON p.product_id = r.product_id
GROUP BY p.product_id, p.title
HAVING COUNT(*) > 0
ORDER BY 平均評分 ASC
LIMIT 1;
```
